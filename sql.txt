CREATE TABLE siswa (
    nisn VARCHAR(10) PRIMARY KEY, -- Pengganti peserta_didik_id bawaan Dapodik
    nipd VARCHAR(20) NOT NULL UNIQUE, -- Nomor Induk Peserta Didik (Nomor Induk Sekolah)
    nik VARCHAR(16) NOT NULL UNIQUE,
    nama_lengkap VARCHAR(150) NOT NULL,
    jenis_kelamin ENUM('L', 'P') NOT NULL,
    tempat_lahir VARCHAR(100),
    tanggal_lahir DATE,
    agama VARCHAR(30),
    anak_keberapa INT DEFAULT 1,
    tinggi_badan INT DEFAULT 0, -- Dalam satuan cm
    berat_badan INT DEFAULT 0,  -- Dalam satuan kg
    no_hp VARCHAR(20) NULL,
    email VARCHAR(100) NULL,
    status_siswa ENUM('Aktif', 'Lulus', 'Pindah', 'Keluar') DEFAULT 'Aktif',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE siswa_orangtua (
    nisn VARCHAR(10) PRIMARY KEY,
    nama_ayah VARCHAR(150) NULL,
    pekerjaan_ayah VARCHAR(100) NULL, -- Diambil dari pekerjaan_ayah_id_str
    nama_ibu VARCHAR(150) NULL,
    pekerjaan_ibu VARCHAR(100) NULL,  -- Diambil dari pekerjaan_ibu_id_str
    nama_wali VARCHAR(150) NULL,
    pekerjaan_wali VARCHAR(100) NULL, -- Diambil dari pekerjaan_wali_id_str
    FOREIGN KEY (nisn) REFERENCES siswa(nisn) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE siswa_registrasi (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nisn VARCHAR(10) NOT NULL,
    registrasi_id VARCHAR(50) NOT NULL, -- ID registrasi unik asli dari Dapodik
    jenis_pendaftaran VARCHAR(50), -- Diambil dari jenis_pendaftaran_id_str (misal: "Siswa baru")
    tanggal_masuk_sekolah DATE,
    sekolah_asal VARCHAR(150),
    nama_rombel VARCHAR(20), -- Contoh: "8 E"
    semester_id VARCHAR(10), -- Contoh: "20252"
    rombongan_belajar_id VARCHAR(50), -- ID Rombel asli Dapodik (berguna jika nanti butuh sync absen per kelas)
    kurikulum_nama VARCHAR(100), -- Diambil dari kurikulum_id_str (misal: "Kurikulum SMP Merdeka")
    FOREIGN KEY (nisn) REFERENCES siswa(nisn) ON DELETE CASCADE,
    -- Mencegah duplikasi data registrasi siswa untuk semester yang sama
    UNIQUE KEY nisn_per_semester (nisn, semester_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
